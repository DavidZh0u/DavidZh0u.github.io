    # 准备Git服务器

    我这里用192.168.163.105做Git服务器 1.安装Git

    [root@host105 ~]# yum install git -y
    1

    2.清一下环境规则

    [root@host105 ~]# iptables -F
    1

    3.创建一个用户

    [root@host105 ~]# useradd git

    [root@host105 ~]# su - git

[root@host105 ~]# su - git

# 如果不存在 .ssh 请先创建
[git@host105 ~]$ chmod 700 .ssh/

# 创建文件authorized_keys，并把上一步的秘钥串复制粘贴其中
[git@host105 ~]$ vi .ssh/authorized_keys

# 修改这个文件的权限 
[git@host105 ~]$ chmod 600 .ssh/authorized_keys
