    3.启动jenkins

    # 先确保已安装JDK
[root@host101 ~]# java -version
java version "1.8.0_231"
Java(TM) SE Runtime Environment (build 1.8.0_231-b11)
Java HotSpot(TM) 64-Bit Server VM (build 25.231-b11, mixed mode)

# 启动
[root@host101 ~]# systemctl start jenkins
# 如果出现这种错误，是因为jenkins调用java路径与我们配置的java环境变量不一致
Job for jenkins.service failed because the control process exited with error code. See "systemctl status jenkins.service" and "journalctl -xe" for details.

[root@host101 ~]# echo $JAVA_HOME
/usr/local/java/jdk1.8.0_231

# 编辑下面这个文件，并把上面的地址加上“/bin/java”后，加入到文件指定Java路径配置中去
[root@host101 ~]# vi /etc/init.d/jenkins

# 再次启动就没有报错了

# 查看jenkins进程
[root@host101 ~]# ps -ef | grep jenkins
# 查看jenkins端口8080
[root@host101 ~]# netstat -antp 8080
# 查看jenkins日志
[root@host101 ~]# tail /var/log/jenkins/jenkins.log -f
