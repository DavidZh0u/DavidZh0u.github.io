# Centos7下安装docker

# 安装步骤

# 1.关闭防火墙

# 关闭防火墙
[root@host106 ~]# systemctl stop firewalld
# 禁止启动开启防火墙
[root@host106 ~]# systemctl disable firewalld
1
2
3
4

# 2.关闭SELinux

[root@host106 ~]# vi /etc/selinux/config
######################################################
# 修改selinux为disabled
# SELINUX=disabled
######################################################

# 重启系统，验证关闭
[root@host106 ~]# getenforce
Disabled

# 3.安装所需的软件包

[root@host106 ~]# yum install -y yum-utils \
                  device-mapper-persistent-data \
                  lvm2
1
2
3

# 4.配置docker-ce的源

[root@host106 ~]# yum-config-manager \
                  --add-repo \
                  https://download.docker.com/linux/centos/docker-ce.repo
1
2
3
4

# 5.安装docker-ce

[root@host106 ~]# yum install docker-ce -y
1

# 6.启动docker,并设置开机启动

[root@host106 ~]# systemctl start docker
[root@host106 ~]# systemctl enable docker
Created symlink from /etc/systemd/system/multi-user.target.wants/docker.service to /usr/lib/systemd/system/docker.service.

# 7.运行hello-world

[root@host106 ~]# docker run hello-world

# 8.运行一个nginx测试

# 运行完，不要ctrl+c，否则会关闭运行，这是一个前台运行
[root@host106 ~]# docker run -it nginx

# 复制会话，查看一下docker进程，可以看到ngnix已经在运行状态"up"
