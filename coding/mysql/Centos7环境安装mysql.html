# 启动mysql

# 不过我们还是常规操作：先设置账号和密码
# 会出现一条警告，无视它
[root@host102 mydir]# /usr/bin/mysqladmin -u root password 'admin321'

# 注意一点：-p后面直接拼密码，没有空格

# 设置远程连接

# 远程连接需要授权
# 我们演示用是全部权限授权 ALL PRIVILEGES
# 如下是给root账户和admin321密码匹配的账户授权
mysql> GRANT ALL PRIVILEGES ON *.* TO 'root'@'%' IDENTIFIED BY 'admin321' WITH GRANT OPTION;
Query OK, 0 rows affected (0.00 sec)

# 如果你希望远程连接可用，要么关闭防火墙，要么暴露监听端口号
# 我这里是暴露mysql的端口号3306
[root@host102 mydir]# firewall-cmd --zone=public --add-port=3306/tcp --permanent  
success
# 重启防火墙
[root@host102 mydir]# firewall-cmd --reload  
success
# 查看3306端口号是否已经暴露出来
[root@host102 mydir]# firewall-cmd --list-ports  
3306/tcp
