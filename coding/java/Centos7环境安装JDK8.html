# 解压JDK文件

# 将压缩包复制到指定文件目录
[root@host102 mydir]# cp jdk-8u231-linux-x64.tar.gz /usr/local/java/

# 解压文件
[root@host102 java]# tar zxvf jdk-8u231-linux-x64.tar.gz

# 移除压缩包
[root@host102 java]# rm  -rf jdk-8u231-linux-x64.tar.gz

# 设置JDK环境变量

# 修改etc/profile
[root@host102 java]# vi /etc/profile
# 文件末尾加上三行
-----------------------------------------------------------------------------
#############java environment =====================
export JAVA_HOME=/usr/local/java/jdk1.8.0_231
export CLASSPATH=.:$JAVA_HOME/jre/lib/rt.jar:$JAVA_HOME/lib/dt.jar:$JAVA_HOME/lib/tools.jar
export PATH=$PATH:$JAVA_HOME/bin
-----------------------------------------------------------------------------

# 使环境变量生效
[root@host102 java]# source  /etc/profile
