# 修改redis.conf
# 修改daemonize的值为yes
# 修改bind 127.0.0.1 的ip地址为本机真实ip:我这台机ip:192.168.163.102
[root@host102 bin]# vi redis.conf 

------------------------------------------------------------------------
...

# JUST COMMENT THE FOLLOWING LINE.
# ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~

...
# By default Redis does not run as a daemon. Use 'yes' if you need it.
# Note that Redis will write a pid file in /var/run/redis.pid when daemonized.
daemonize yes
...
------------------------------------------------------------------------

# 远程客户端连接

# 修改redis.conf文件
# 注释掉bind 192.168.163.102
# 将protect-mode的值设置为 no
# 设置密码admin321
------------------------------------------------------------------------
#bind 192.168.163.102	
protected-mode no
requirepass admin321
------------------------------------------------------------------------

# 重新启动redis服务
# 暴露6379端口给外部
[root@host102 bin]# firewall-cmd --zone=public --add-port=6379/tcp --permanent
success
[root@host102 bin]# firewall-cmd --reload 
success
[root@host102 bin]# firewall-cmd --list-ports
3306/tcp 6379/tcp
